Why OOP?

Start with a relatable analogy. A class is a blueprint (like an architectural plan). An object is a house built from that plan. 100 houses can be built from one plan — each with its own color, owner, and furniture.

Procedural vs OOP: Procedural code is a list of instructions. OOP organises code around things (objects) that have state (attributes) and behaviour (methods). Easier to extend, reuse, and maintain.

Classes and Objects

A class defines structure. An object is an instance of that class. __init__ is the constructor — it runs when an object is created. self refers to the current object instance.

example_1_class_object.py Python 3
# Define a class
class Student:
    def __init__(self, name, roll_no):
        self.name    = name        # instance attribute
        self.roll_no = roll_no
        self.marks   = []          # starts empty for every student

# Create objects (instances)
s1 = Student("Aanya", 101)
s2 = Student("Rohan", 102)

print(s1.name)      # Aanya
print(s2.roll_no)   # 102
print(s1 is s2)     # False — they are different objects

Topic 2 · 8 min

Instance vs Class Variables

Instance variables are unique per object (set via self). Class variables are shared across all instances of the class.

example_2_variables.py
class Student:
    school_name = "IIT Delhi"    # class variable — shared by ALL students
    total_students = 0

    def __init__(self, name, roll_no):
        self.name    = name          # instance variable — unique per student
        self.roll_no = roll_no
        Student.total_students += 1  # update class variable on each new object

s1 = Student("Aanya", 101)
s2 = Student("Rohan", 102)

print(s1.school_name)           # IIT Delhi
print(Student.school_name)      # IIT Delhi  (same thing)
print(Student.total_students)   # 2

# Modifying class variable via instance creates a NEW instance variable — tricky!
s1.school_name = "NIT Trichy"
print(s1.school_name)           # NIT Trichy  (instance's own copy)
print(s2.school_name)           # IIT Delhi   (class variable unchanged)
Common trap: Mutable class variables (lists, dicts) are shared — modifying them on one instance affects all. Demo this live to drive the point home.
Topic 3 · 10 min

Instance, Class, and Static Methods

example_3_methods.py
class Student:
    school_name = "IIT Delhi"

    def __init__(self, name, roll_no):
        self.name    = name
        self.roll_no = roll_no
        self.marks   = []

    # ── Instance method: operates on a specific object
    def add_mark(self, mark):
        self.marks.append(mark)

    def average(self):
        if not self.marks:
            return 0
        return sum(self.marks) / len(self.marks)

    # ── Class method: operates on the class itself (not one object)
    @classmethod
    def change_school(cls, new_name):
        cls.school_name = new_name

    # ── Static method: utility — no access to self or cls
    @staticmethod
    def is_passing(mark):
        return mark >= 40


s1 = Student("Aanya", 101)
s1.add_mark(78)
s1.add_mark(85)
print(s1.average())                    # 81.5
print(Student.is_passing(35))          # False
Student.change_school("NIT Surathkal")
print(s1.school_name)                  # NIT Surathkal
Topic 4 · 7 min

Dunder (Magic) Methods

Python calls these automatically in specific situations. They let your class integrate naturally with Python syntax.

example_4_dunder.py
class Student:
    def __init__(self, name, roll_no):
        self.name    = name
        self.roll_no = roll_no
        self.marks   = []

    # Called by print() and str()
    def __str__(self):
        return f"Student({self.name}, Roll: {self.roll_no})"

    # Called in the REPL / repr() — for developers
    def __repr__(self):
        return f"Student(name={self.name!r}, roll_no={self.roll_no!r})"

    # Called by len()
    def __len__(self):
        return len(self.marks)

    # Called by == operator
    def __eq__(self, other):
        return self.roll_no == other.roll_no


s1 = Student("Aanya", 101)
s1.marks = [78, 85, 91]
print(s1)           # Student(Aanya, Roll: 101)
print(len(s1))      # 3  (number of marks)
s2 = Student("Rohan", 101)
print(s1 == s2)     # True  (same roll number)

class Student:
    school_name    = "My College"
    total_students = 0

    def __init__(self, name, roll_no):
        self.name    = name
        self.roll_no = roll_no
        self.marks   = []
        Student.total_students += 1

    def add_mark(self, subject, mark):
        self.marks.append({"subject": subject, "mark": mark})

    def average(self):
        if not self.marks: return 0.0
        return sum(m["mark"] for m in self.marks) / len(self.marks)

    def grade(self):
        avg = self.average()
        if   avg >= 90: return "A+"
        elif avg >= 75: return "A"
        elif avg >= 60: return "B"
        elif avg >= 40: return "C"
        else:          return "F"

    def report(self):
        lines = [f"=== Report: {self.name} (Roll {self.roll_no}) ==="]
        for m in self.marks:
            lines.append(f"  {m['subject']}: {m['mark']}")
        lines.append(f"  Average: {self.average():.1f}  |  Grade: {self.grade()}")
        return "\n".join(lines)

    def __str__(self):
        return f"Student({self.name}, {self.grade()})"

Encapsulation

Bundling data + methods together and controlling access. Python uses naming conventions (no true private), but the intent is enforced by convention and properties.

example_5_encapsulation.py
class Student:
    def __init__(self, name, roll_no):
        self.name     = name          # public — access freely
        self._section = "A"          # protected — internal use, but accessible
        self.__roll   = roll_no       # private — name-mangled to _Student__roll
        self.__marks  = []

    # Property: looks like attribute, behaves like method
    @property
    def roll_no(self):
        return self.__roll

    @roll_no.setter
    def roll_no(self, value):
        if not (100 <= value <= 999):
            raise ValueError("Roll number must be 100–999")
        self.__roll = value

    def add_mark(self, mark):
        if not (0 <= mark <= 100):
            raise ValueError(f"Invalid mark: {mark}")
        self.__marks.append(mark)

    def get_marks(self):
        return tuple(self.__marks)   # return copy, not the list itself


s = Student("Aanya", 101)
print(s.roll_no)        # 101  (via property getter)
s.roll_no = 205          # calls setter — validates!
s.add_mark(88)
# s.__marks             # AttributeError — good!
print(s.get_marks())    # (88,)

try:
    s.roll_no = 5        # triggers validation
except ValueError as e:
    print(e)             # Roll number must be 100–999
Topic 2 · 12 min

Inheritance

A child class inherits all attributes and methods of its parent. Use super() to call the parent's __init__. Child can add new methods or override existing ones.

example_6_inheritance.py
# ── Base class
class Student:
    def __init__(self, name, roll_no):
        self.name    = name
        self.roll_no = roll_no
        self.marks   = []

    def add_mark(self, mark): self.marks.append(mark)
    def average(self):        return sum(self.marks)/len(self.marks) if self.marks else 0
    def describe(self):        return f"UG Student: {self.name}"


# ── Derived class: inherits from Student
class GradStudent(Student):
    def __init__(self, name, roll_no, thesis_title):
        super().__init__(name, roll_no)   # call parent __init__
        self.thesis = thesis_title
        self.thesis_submitted = False

    # Override the parent method
    def describe(self):
        return f"PG Student: {self.name} | Thesis: {self.thesis}"

    # New method specific to GradStudent
    def submit_thesis(self):
        self.thesis_submitted = True
        return f"{self.name}'s thesis submitted!"


# ── Multilevel: PhD inherits from GradStudent
class PhDStudent(GradStudent):
    def __init__(self, name, roll_no, thesis_title, guide):
        super().__init__(name, roll_no, thesis_title)
        self.guide = guide

    def describe(self):
        return f"PhD: {self.name} | Guide: {self.guide}"


# --- Test ---
ug  = Student("Aanya", 101)
pg  = GradStudent("Rohan", 201, "ML in Healthcare")
phd = PhDStudent("Priya", 301, "Quantum ML", "Dr. Sharma")

print(ug.describe())      # UG Student: Aanya
print(pg.describe())      # PG Student: Rohan | Thesis: ML in Healthcare
print(phd.describe())     # PhD: Priya | Guide: Dr. Sharma

pg.add_mark(88)           # inherited from Student
print(pg.average())       # 88.0

print(isinstance(phd, Student))   # True — PhDStudent IS-A Student
Topic 3 · 10 min

Polymorphism

"Many forms." The same method name behaves differently depending on the object. Python achieves this through method overriding and duck typing — if it has the right method, Python doesn't care about the type.

example_7_polymorphism.py
# ── Method overriding (runtime polymorphism)
def print_info(person):
    # Works for Student, GradStudent, PhDStudent — any object with .describe()
    print(person.describe())

students = [
    Student("Aanya", 101),
    GradStudent("Rohan", 201, "AI Ethics"),
    PhDStudent("Priya", 301, "Quantum ML", "Dr. Sharma")
]

for s in students:
    print_info(s)   # same call, different behaviour

# Output:
# UG Student: Aanya
# PG Student: Rohan | Thesis: AI Ethics
# PhD: Priya | Guide: Dr. Sharma


# ── Duck typing — "If it walks like a duck, it's a duck"
class Professor:
    def __init__(self, name): self.name = name
    def describe(self):         return f"Professor: {self.name}"

# Professor is NOT a Student, but print_info still works!
print_info(Professor("Dr. Mehta"))   # Professor: Dr. Mehta
Key idea: Python doesn't do type checks — it checks for the method's existence at runtime. This is why Python functions are flexible by design.

class GradStudent(Student):      # Student from Session 1
    _MAX_CHAPTERS = 8

    def __init__(self, name, roll_no, thesis_title, advisor):
        super().__init__(name, roll_no)
        self.__thesis  = thesis_title
        self.advisor   = advisor
        self.__chapters = 0

    @property
    def chapters_done(self):
        return self.__chapters

    def complete_chapter(self):
        if self.__chapters >= self._MAX_CHAPTERS:
            raise Exception("All chapters already done!")
        self.__chapters += 1
        return f"Chapter {self.__chapters} done. {self._MAX_CHAPTERS - self.__chapters} remaining."

    def progress(self):
        pct = (self.__chapters / self._MAX_CHAPTERS) * 100
        bar = "█" * self.__chapters + "░" * (self._MAX_CHAPTERS - self.__chapters)
        return f"{bar} {pct:.0f}%"

    def describe(self):
        return f"PG: {self.name} | Thesis: {self.__thesis} | {self.progress()}"

Abstraction with Abstract Base Classes

Abstraction hides implementation details and exposes only what's necessary. ABC forces subclasses to implement specific methods — you can't skip them.

example_8_abstraction.py
from abc import ABC, abstractmethod

class Person(ABC):
    def __init__(self, name):
        self.name = name

    @abstractmethod
    def describe(self):
        """Every subclass MUST implement this."""
        pass

    @abstractmethod
    def calculate_fee(self):
        pass

    # Concrete method — shared by all subclasses
    def greet(self):
        return f"Hello, I am {self.name}."


class Student(Person):
    def __init__(self, name, program):
        super().__init__(name)
        self.program = program

    def describe(self):
        return f"Student: {self.name} | {self.program}"

    def calculate_fee(self):
        return 80_000


class Professor(Person):
    def describe(self):
        return f"Professor: {self.name}"

    def calculate_fee(self):
        return 0   # no fee for faculty


# Person() → TypeError: Can't instantiate abstract class
s = Student("Aanya", "B.Tech CSE")
p = Professor("Dr. Mehta")
print(s.describe())        # Student: Aanya | B.Tech CSE
print(s.calculate_fee())   # 80000
print(p.greet())           # Hello, I am Dr. Mehta.  (concrete method inherited)
Topic 2 · 7 min

Multiple Inheritance & MRO

Python supports inheriting from multiple parents. The Method Resolution Order (MRO) — determined by the C3 linearisation algorithm — defines the lookup order. Use ClassName.__mro__ to inspect it.

example_9_mro.py
class Researcher:
    def publish(self):
        return f"{self.name} published a paper"

    def role(self):
        return "Researcher"


class TeachingAssistant:
    def conduct_lab(self):
        return f"{self.name} is conducting a lab"

    def role(self):
        return "TA"


# Multiple inheritance: ResearchScholar inherits from BOTH
class ResearchScholar(Student, Researcher, TeachingAssistant):
    def __init__(self, name, roll_no):
        Student.__init__(self, name, roll_no)   # explicit call


rs = ResearchScholar("Priya", 501)
print(rs.publish())        # Priya published a paper
print(rs.conduct_lab())    # Priya is conducting a lab
print(rs.role())           # "Student" — MRO picks Student.role() first? No — Student
                            # doesn't define .role(), so Researcher wins

# Inspect the lookup order
print([c.__name__ for c in ResearchScholar.__mro__])
# ['ResearchScholar', 'Student', 'Researcher', 'TeachingAssistant', 'object']
Rule of thumb: Prefer composition over multiple inheritance when possible. Use multiple inheritance only for true "is-a-kind-of-both" relationships.

class Classroom:
    def __init__(self, section, max_strength=60):
        self.section      = section
        self.max_strength = max_strength
        self.__students   = []

    def add_student(self, student):
        if len(self.__students) >= self.max_strength:
            raise Exception("Classroom is full!")
        if self.get_student(student.roll_no):
            raise ValueError(f"Roll {student.roll_no} already exists")
        self.__students.append(student)

    def get_student(self, roll_no):
        return next((s for s in self.__students if s.roll_no == roll_no), None)

    def class_average(self):
        avgs = [s.average() for s in self.__students if s.marks]
        return sum(avgs) / len(avgs) if avgs else 0.0

    def topper(self):
        students_with_marks = [s for s in self.__students if s.marks]
        if not students_with_marks: return None
        return max(students_with_marks, key=lambda s: s.average())

    def full_report(self):
        print(f"\n{'='*50}")
        print(f"  CLASSROOM {self.section} REPORT")
        print(f"{'='*50}")
        for s in self.__students:
            print(s.report())
            print("-"*40)
        t = self.topper()
        print(f"\n🏆 Topper: {t.name} ({t.average():.1f})" if t else "No marks yet")
        print(f"📊 Class Average: {self.class_average():.1f}")

    def __len__(self):  return len(self.__students)
    def __str__(self):  return f"Classroom {self.section} | {len(self)} students"

Wrap-up · 2:50 – 3:00

  • Class = blueprint, Object = instance. __init__ and self are always there
  • Encapsulation: use __private + @property to control access
  • Inheritance: super() chains back to parent; child can override any method
  • Polymorphism: same interface, different behaviour — the power of OOP
  • Abstraction: ABC + @abstractmethod enforces contracts in your hierarchy
  • MRO determines method lookup in multiple inheritance — always check with __mro__
